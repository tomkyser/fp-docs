{{- /* Clean URL render hook — strips .md extensions from internal links */ -}} {{- $url := .Destination -}} {{- if and (not (strings.HasPrefix $url "http")) (strings.HasSuffix $url ".md") -}} {{- $url = strings.TrimSuffix ".md" $url -}} {{- $url = printf "%s/" $url -}} {{- end -}} {{- /* Handle _index.md references */ -}} {{- if strings.HasSuffix $url "/_index/" -}} {{- $url = strings.TrimSuffix "_index/" $url -}} {{- end -}} {{- .Text | safeHTML -}}
